   \begin{align*}
      ||A-B||=\sqrt{(A-B)^\top(A-B)}\\
      \vec{A}=\myvec{4\\2}   \vec{B}=\myvec{8\\4}\\
      (A-B)=\myvec{-4\\-2}\\

   \end{align*}

   \begin{align*}
      AB=\sqrt{\myvec{-4 -2} \myvec{-4\\-2}} \\=2\sqrt{5}\\
   \end{align*}
   
   Similarly,

   \begin{align*}
      AP=\sqrt{\myvec{-2 -1} \myvec{-2\\-1}}\\=\sqrt{5}\\
      PB=\sqrt{\myvec{-6 -3} \myvec{-6\\-3}}\\=3\sqrt{5}\\
   \end{align*}

 Therefore,
   \begin{align*}
     AP=\frac{1}{2}AB
   \end{align*}
